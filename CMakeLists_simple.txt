cmake_minimum_required(VERSION 3.20)
project(pebbl)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# For now, let's comment out boost dependency and use a simple integer type
# We'll create a simple replacement for boost::multiprecision::cpp_int

set(SOURCES
  src/main.cpp
  src/parser/lexer/lexer.cpp
  src/parser/ast_generation/ast_generator.cpp
  src/runtime/gc.cpp
  src/runtime/object/object.cpp
  src/runtime/evaluator/environment.cpp
  src/runtime/evaluator/interpreter.cpp
  src/runtime/bytecode/bytecode.cpp
  src/runtime/bytecode/compiler.cpp
  src/runtime/bytecode/vm.cpp
  src/runtime/builtins/
)

set(HEADER_DIRS
  src/parser/lexer
  src/parser/ast_generation
  src/runtime
  src/runtime/object
  src/runtime/evaluator
  src/runtime/bytecode
  src/runtime/builtins
  src/
)

if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Debug CACHE STRING "Build type")
endif()

add_executable(pebbli ${SOURCES})
target_include_directories(pebbli PRIVATE ${HEADER_DIRS})

if (MSVC)
  target_compile_options(pebbli PRIVATE  
    /W4
    $<$<CONFIG:Release>:/O2>
    $<$<CONFIG:Debug>:/Od>
    $<$<CONFIG:Debug>:/Zi>
  )
else()
  target_compile_options(pebbli PRIVATE
    -Wall
    -Wextra
    -Wpedantic
    $<$<CONFIG:Release>:-O3>
    $<$<CONFIG:Debug>:-O0>
    $<$<CONFIG:Debug>:-g>
  )
endif()

install(TARGETS pebbli RUNTIME DESTINATION bin)
