82inline TokenType lookup_identifier(const std::string& name) {
83 static const std::unordered_map<std::string, TokenType> keywords = {
84 {"and", TokenType::AND},
85 {"or", TokenType::OR},
86 {"if", TokenType::IF},
87 {"else", TokenType::ELSE},
88 {"true", TokenType::TRUE},
89 {"false", TokenType::FALSE},
90 {"for", TokenType::FOR},
91 {"while", TokenType::WHILE},
92 {"func", TokenType::FUNC},
93 {"return", TokenType::RETURN},
94 {"let", TokenType::LET},
95 {"var", TokenType::VAR},
96 {"nil", TokenType::NIL}};
97 if (auto it = keywords.find(name); it != keywords.end()) {
98 return it->second;
99 } else {
100 return TokenType::IDENTIFIER;
101 }
